js/binaryCalculator.js function calculateResult(res) { var text = res.innerText; var op = text.match(/\+|-|\*|\//g)[0]; var arr = text.split(op); var operand1 = parseInt(arr[0], 2); var operand2 = parseInt(arr[1], 2); var result; switch(op) { case "+": result = operand1 + operand2; break; case "-": result = operand1 - operand2; break; case "*": result = operand1 * operand2; break; case "/": result = operand1 / operand2; break; } res.innerText = result.toString(2); } var action = function() { var e = window.event; var btn = e.target || e.srcElement; var res = document.getElementById("res"); switch(btn.id) { case "btn0": res.innerHTML = res.innerHTML + "0"; break; case "btn1": res.innerHTML = res.innerHTML + "1"; break; case "btnClr": res.innerHTML = ""; break; case "btnEql": calculateResult(res); break; case "btnSum": res.innerHTML = res.innerHTML + "+"; break; case "btnSub": res.innerHTML = res.innerHTML + "-"; break; case "btnMul": res.innerHTML = res.innerHTML + "*"; break; case "btnDiv": res.innerHTML = res.innerHTML + "/"; break; } } document.getElementById("btn0").onclick = action; document.getElementById("btn1").onclick = action; document.getElementById("btnClr").onclick = action; document.getElementById("btnEql").onclick = action; document.getElementById("btnSum").onclick = action; document.getElementById("btnSub").onclick = action; document.getElementById("btnMul").onclick = action; document.getElementById("btnDiv").onclick = action; css/binaryCalculator.css #res { background-color: lightgray; border: solid; width: 81%; height: 48px; font-size: 20px; } #btns { width: 90%; } #btn0, #btn1 { background-color: lightgreen; color: brown; } #btnClr, #btnEql { background-color: darkgreen; color: white; } #btnSum, #btnSub, #btnMul, #btnDiv { background-color: black; color: red; } .buttonClass { width: 22%; height: 36px; font-size: 18px; }
